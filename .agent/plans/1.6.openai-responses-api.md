# 1.6 OpenRouter Chat Completions Integration

⚠️ **Medium** - May need iteration, some complexity

## Context
Replaces the stub streaming generator in `backend/routers/chat.py` with real OpenRouter Chat Completions API calls. OpenRouter exposes an OpenAI-compatible API — we use the `openai` Python SDK with a custom `base_url`.

**Threading model (stateless):**
- No provider-side thread objects. Conversation continuity is achieved by loading all prior messages for the thread from Supabase and sending them in the `messages` array with every request.
- The `openai_thread_id` column on `threads` is unused.

**Free model:** `google/gemini-2.0-flash-exp:free` — change in `OPENROUTER_MODEL` env var to swap models without code changes.

## Tasks

### OpenRouter Service
1. Create `backend/services/__init__.py` (empty)

2. Create `backend/services/llm_service.py`:
   ```python
   from typing import Generator
   from openai import OpenAI
   from config import settings

   client = OpenAI(
       base_url="https://openrouter.ai/api/v1",
       api_key=settings.OPENROUTER_API_KEY,
   )

   SYSTEM_PROMPT = "You are a helpful AI assistant. Answer questions clearly and concisely."
   MODEL = "google/gemini-2.0-flash-exp:free"

   def stream_chat(history: list[dict]) -> Generator[str, None, None]:
       """
       history: list of {"role": "user"|"assistant", "content": str}
       Yields text deltas as they arrive from OpenRouter.
       """
       messages = [{"role": "system", "content": SYSTEM_PROMPT}] + history
       with client.chat.completions.create(
           model=MODEL,
           messages=messages,
           stream=True,
       ) as stream:
           for chunk in stream:
               delta = chunk.choices[0].delta.content
               if delta:
                   yield delta
   ```

### Updated Chat Router
3. Update `backend/routers/chat.py` to replace `stub_generator` with real OpenRouter calls:
   ```python
   import json
   from fastapi import APIRouter, Depends, HTTPException
   from fastapi.responses import StreamingResponse
   from dependencies import get_current_user
   from database import supabase
   from models import ChatRequest
   from services.llm_service import stream_chat

   router = APIRouter(prefix="/api/chat", tags=["chat"])

   async def chat_generator(body: ChatRequest, user_id: str):
       # Resolve or create thread
       if body.thread_id:
           result = supabase.table("threads") \
               .select("id, title") \
               .eq("id", body.thread_id) \
               .eq("user_id", user_id) \
               .single() \
               .execute()
           if not result.data:
               raise HTTPException(status_code=404, detail="Thread not found")
           thread_id = result.data["id"]
       else:
           title = body.message[:60] + ("..." if len(body.message) > 60 else "")
           result = supabase.table("threads") \
               .insert({"user_id": user_id, "title": title}) \
               .execute()
           thread_id = result.data[0]["id"]

       # Persist user message
       supabase.table("messages").insert({
           "thread_id": thread_id,
           "user_id": user_id,
           "role": "user",
           "content": body.message,
       }).execute()

       # Load full message history for this thread
       history_result = supabase.table("messages") \
           .select("role, content") \
           .eq("thread_id", thread_id) \
           .order("created_at") \
           .execute()
       history = [{"role": m["role"], "content": m["content"]} for m in history_result.data]

       # Stream from OpenRouter
       full_content = ""
       for delta in stream_chat(history):
           full_content += delta
           chunk = {"type": "chunk", "content": delta}
           yield f"data: {json.dumps(chunk)}\n\n"

       # Persist assistant message
       msg_result = supabase.table("messages").insert({
           "thread_id": thread_id,
           "user_id": user_id,
           "role": "assistant",
           "content": full_content,
       }).execute()
       message_id = msg_result.data[0]["id"]

       done = {"type": "done", "thread_id": thread_id, "message_id": message_id}
       yield f"data: {json.dumps(done)}\n\n"

   @router.post("")
   async def chat(body: ChatRequest, user=Depends(get_current_user)):
       return StreamingResponse(
           chat_generator(body, user.id),
           media_type="text/event-stream",
           headers={"Cache-Control": "no-cache", "X-Accel-Buffering": "no"},
       )
   ```
